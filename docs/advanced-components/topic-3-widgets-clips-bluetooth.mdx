## WidgetKit — timelines, not live views

<Callout type="warning" title="Widgets are snapshots, not live">
  A widget is not a running view. It's a series of pre-rendered entries on a
  <em>timeline</em> that iOS picks from at display time. Your widget's
  <code>body</code> runs when the system renders a snapshot — not on a timer,
  not in response to data changes. If you need live behavior, use a Live
  Activity (ActivityKit) or a <code>deeplink</code> to the app. Expect
  refreshes no more often than every ~15 minutes on battery.
</Callout>

### TimelineProvider vs IntentTimelineProvider

Use `TimelineProvider` for widgets that need no configuration. Use the modern
`AppIntentTimelineProvider` (iOS 17+) for widgets where the user picks what
they see — a stock symbol, a city, a specific list.

```swift
import WidgetKit
import SwiftUI
import AppIntents

struct SymbolEntry: TimelineEntry {
    let date: Date
    let symbol: String
    let price: Decimal
}

struct PickSymbolIntent: WidgetConfigurationIntent {
    static var title: LocalizedStringResource = "Pick a symbol"
    @Parameter(title: "Symbol", default: "AAPL") var symbol: String
}

struct SymbolProvider: AppIntentTimelineProvider {
    func placeholder(in context: Context) -> SymbolEntry {
        SymbolEntry(date: .now, symbol: "AAPL", price: 0)
    }

    func snapshot(for configuration: PickSymbolIntent,
                  in context: Context) async -> SymbolEntry {
        let price = (try? await Market.price(for: configuration.symbol)) ?? 0
        return SymbolEntry(date: .now, symbol: configuration.symbol, price: price)
    }

    func timeline(for configuration: PickSymbolIntent,
                  in context: Context) async -> Timeline<SymbolEntry> {
        var entries: [SymbolEntry] = []
        for offset in 0..<6 {
            let date = Date.now.addingTimeInterval(TimeInterval(offset * 15 * 60))
            let price = (try? await Market.price(for: configuration.symbol)) ?? 0
            entries.append(SymbolEntry(date: date, symbol: configuration.symbol, price: price))
        }
        return Timeline(entries: entries, policy: .after(.now.addingTimeInterval(90 * 60)))
    }
}

struct SymbolWidgetView: View {
    let entry: SymbolEntry
    var body: some View {
        VStack(alignment: .leading) {
            Text(entry.symbol).font(.headline)
            Text(entry.price, format: .currency(code: "USD"))
                .font(.system(.title2, design: .rounded, weight: .semibold))
                .contentTransition(.numericText())
        }
        .containerBackground(.fill.tertiary, for: .widget)
    }
}

struct SymbolWidget: Widget {
    var body: some WidgetConfiguration {
        AppIntentConfiguration(kind: "Symbol",
                               intent: PickSymbolIntent.self,
                               provider: SymbolProvider()) { entry in
            SymbolWidgetView(entry: entry)
        }
        .supportedFamilies([.systemSmall, .accessoryRectangular, .accessoryInline])
    }
}
```

Tap handling is via `Link` + a `URL` that your app opens, or — iOS 17+ —
via `Button(intent:)` with an `AppIntent` that can mutate state and call
`WidgetCenter.shared.reloadAllTimelines()`.

## Live Activities — the one live thing

When you genuinely need real-time UI on the Lock Screen and Dynamic Island
(delivery tracking, sports scores, timers), that's ActivityKit, not WidgetKit.

```swift
import ActivityKit

struct DeliveryAttributes: ActivityAttributes {
    struct ContentState: Codable, Hashable {
        var etaMinutes: Int
        var status: String
    }
    let orderID: String
}

let attrs = DeliveryAttributes(orderID: "A-812")
let state = DeliveryAttributes.ContentState(etaMinutes: 12, status: "On the way")
let content = ActivityContent(state: state, staleDate: .now.addingTimeInterval(60 * 60))
let activity = try Activity.request(attributes: attrs, content: content, pushType: .token)
```

Updates come from your server via a push token you register for the activity,
or locally via `activity.update(_:)`. UI is a WidgetKit bundle with a
`ActivityConfiguration` entry.

## App Clips — your app, without the install

An App Clip is a small (&lt; 15 MB uncompressed) slice of your app that launches
from a URL, NFC tag, QR code, App Clip Code, or Messages link. It has access
to a limited set of frameworks — no HealthKit, no background tasks — and
presents a single focused task (pay for parking, order this coffee).

### Clip → full app migration

The clip and full app share an App Group, so any data the clip writes
(pending order, partial profile) is available when the user installs the
full app. Use `ASWebAuthenticationSession` for Sign in with Apple at My Apple
ID — it persists across the install. Design for the clip to be *useful even
if the user never installs*, and *delightful if they do*.

### Minimal central

```swift
import CoreBluetooth

final class Scanner: NSObject, CBCentralManagerDelegate, CBPeripheralDelegate {
    private var manager: CBCentralManager!
    private var peripheral: CBPeripheral?
    private let heartRate = CBUUID(string: "180D")
    private let measurement = CBUUID(string: "2A37")

    override init() {
        super.init()
        manager = CBCentralManager(delegate: self, queue: nil)
    }

    func centralManagerDidUpdateState(_ central: CBCentralManager) {
        if central.state == .poweredOn {
            central.scanForPeripherals(withServices: [heartRate])
        }
    }

    func centralManager(_ central: CBCentralManager,
                        didDiscover peripheral: CBPeripheral,
                        advertisementData: [String : Any], rssi RSSI: NSNumber) {
        self.peripheral = peripheral
        peripheral.delegate = self
        central.stopScan()
        central.connect(peripheral)
    }

    func centralManager(_ central: CBCentralManager,
                        didConnect peripheral: CBPeripheral) {
        peripheral.discoverServices([heartRate])
    }

    func peripheral(_ peripheral: CBPeripheral, didDiscoverServices error: Error?) {
        peripheral.services?.forEach {
            peripheral.discoverCharacteristics([measurement], for: $0)
        }
    }

    func peripheral(_ peripheral: CBPeripheral,
                    didDiscoverCharacteristicsFor service: CBService, error: Error?) {
        service.characteristics?.forEach { peripheral.setNotifyValue(true, for: $0) }
    }

    func peripheral(_ peripheral: CBPeripheral,
                    didUpdateValueFor characteristic: CBCharacteristic, error: Error?) {
        guard let data = characteristic.value, let bpm = data.dropFirst().first else { return }
        print("Heart rate: \(bpm) bpm")
    }
}
```

Enable `Background Modes → Uses Bluetooth LE accessories` if you need
connections to survive into the background — the OS will relaunch you for
state restoration if you set `CBCentralManagerOptionRestoreIdentifierKey`.

## Exercises

1. Build a configurable Lock Screen widget (`accessoryRectangular`) that
   shows the next event from a user-selected calendar. Use
   `AppIntentTimelineProvider` + a `WidgetConfigurationIntent` so the user
   picks the calendar from the widget's long-press menu.
2. Write a Core Bluetooth scanner that discovers any peripheral advertising
   the Battery Service (`0x180F`), connects, reads the Battery Level
   characteristic (`0x2A19`), and displays the value. Handle Bluetooth being
   off, unauthorized, and the peripheral disconnecting mid-read.
