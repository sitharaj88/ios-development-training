## Sign in with Apple — the required path

App Store Review Guideline 4.8 makes Sign in with Apple mandatory if you offer
any third-party social login. Wire the Apple credential into Firebase Auth via
`OAuthProvider(providerID: "apple.com")`.

```swift
import AuthenticationServices
import CryptoKit
import FirebaseAuth

final class AppleSignInCoordinator: NSObject, ASAuthorizationControllerDelegate {
    private var currentNonce: String?

    func start() {
        let nonce = Self.randomNonceString()
        currentNonce = nonce
        let request = ASAuthorizationAppleIDProvider().createRequest()
        request.requestedScopes = [.fullName, .email]
        request.nonce = Self.sha256(nonce)
        let controller = ASAuthorizationController(authorizationRequests: [request])
        controller.delegate = self
        controller.performRequests()
    }

    func authorizationController(controller: ASAuthorizationController,
                                 didCompleteWithAuthorization auth: ASAuthorization) {
        guard
            let appleIDCred = auth.credential as? ASAuthorizationAppleIDCredential,
            let tokenData = appleIDCred.identityToken,
            let idToken = String(data: tokenData, encoding: .utf8),
            let nonce = currentNonce
        else { return }

        let credential = OAuthProvider.credential(
            withProviderID: "apple.com",
            idToken: idToken,
            rawNonce: nonce
        )
        Task {
            let result = try await Auth.auth().signIn(with: credential)
            print("Firebase uid:", result.user.uid)
        }
    }
}
```

Google sign-in follows the same pattern via [GoogleSignIn-iOS](https://github.com/google/GoogleSignIn-iOS),
producing an `idToken` + `accessToken` you feed into `GoogleAuthProvider.credential`.
Email/password is `Auth.auth().createUser(withEmail:password:)` and anonymous is
`Auth.auth().signInAnonymously()` — use it for guest checkouts, then *link* the
anonymous user to a real credential so their data survives.

## Firestore — the document database

<FeatureGrid
  features={[
    { title: 'Real-time listeners', description: 'addSnapshotListener pushes updates to every device subscribed to a query in ~100ms.' },
    { title: 'Offline persistence', description: 'Reads and writes work offline; the SDK reconciles when the device reconnects.' },
    { title: 'Transactions', description: 'Read-modify-write with automatic retry on conflicts — safe money math and counters.' },
    { title: 'Batched writes', description: 'Up to 500 writes committed atomically, perfect for fan-out on message send.' },
    { title: 'Compound queries', description: 'where + orderBy + limit, with composite indexes auto-suggested in the console.' },
    { title: 'Security rules', description: 'Expression language running on Google infra — the only trustworthy auth boundary.' },
  ]}
/>

```swift
import FirebaseFirestore

struct Order: Codable, Identifiable {
    @DocumentID var id: String?
    let userId: String
    let total: Decimal
    let createdAt: Date
}

@MainActor
final class OrdersStore: ObservableObject {
    @Published private(set) var orders: [Order] = []
    private var listener: ListenerRegistration?

    func start(for uid: String) {
        listener = Firestore.firestore()
            .collection("orders")
            .whereField("userId", isEqualTo: uid)
            .order(by: "createdAt", descending: true)
            .limit(to: 50)
            .addSnapshotListener { [weak self] snap, _ in
                self?.orders = snap?.documents.compactMap { try? $0.data(as: Order.self) } ?? []
            }
    }

    deinit { listener?.remove() }
}
```

### Transactions and batched writes

```swift
try await Firestore.firestore().runTransaction { txn, errorPointer in
    let ref = Firestore.firestore().collection("counters").document("orders")
    let snap = try? txn.getDocument(ref)
    let current = snap?.data()?["count"] as? Int ?? 0
    txn.updateData(["count": current + 1], forDocument: ref)
    return nil
}

let batch = Firestore.firestore().batch()
batch.setData(message.asDictionary, forDocument: messagesRef.document())
batch.updateData(["lastMessageAt": Date()], forDocument: roomRef)
try await batch.commit()
```

## Security rules — the real perimeter

<Callout kind="danger" title="Never trust client-side security">
  Rules in Swift (hiding buttons, filtering arrays) are UX, not security. Every
  Firestore read and write must be gated by a rule on the server. Assume a
  determined attacker will call the SDK directly with a forged request body.
</Callout>

```
rules_version = '2';
service cloud.firestore {
  match /databases/{db}/documents {
    match /orders/{orderId} {
      allow read: if request.auth != null
                  && request.auth.uid == resource.data.userId;
      allow create: if request.auth != null
                    && request.auth.uid == request.resource.data.userId;
      allow update, delete: if false; // orders are immutable client-side
    }
  }
}
```

Test rules with the [Firebase Local Emulator Suite](https://firebase.google.com/docs/emulator-suite)
before every deploy — the `firebase emulators:exec` command runs a rules test
suite in CI.

display name and avatar URL on each post, but store the canonical user profile
at `/users/{uid}`.

## Offline persistence

```swift
let settings = FirestoreSettings()
settings.cacheSettings = PersistentCacheSettings(
    sizeBytes: NSNumber(value: 100 * 1024 * 1024) // 100 MB
)
Firestore.firestore().settings = settings
```

With persistence on, the SDK serves cached results instantly and queues writes
until the device reconnects. Your SwiftUI views don't know the network is down.
