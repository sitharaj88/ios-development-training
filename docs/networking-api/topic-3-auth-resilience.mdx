## OAuth 2.0 Authorization Code + PKCE

PKCE ("Pixy") is the flavour of OAuth that works without a client secret — which
you can't keep secret on a mobile device. The flow:

### The implementation

```swift
import AuthenticationServices
import CryptoKit

final class OAuthCoordinator: NSObject, ASWebAuthenticationPresentationContextProviding {
    let clientID: String
    let redirect: URL
    let authorize: URL
    let token: URL

    func signIn() async throws -> TokenPair {
        let verifier = Self.codeVerifier()
        let challenge = Self.codeChallenge(for: verifier)
        var comps = URLComponents(url: authorize, resolvingAgainstBaseURL: false)!
        comps.queryItems = [
            .init(name: "response_type", value: "code"),
            .init(name: "client_id", value: clientID),
            .init(name: "redirect_uri", value: redirect.absoluteString),
            .init(name: "scope", value: "openid profile offline_access"),
            .init(name: "code_challenge", value: challenge),
            .init(name: "code_challenge_method", value: "S256"),
            .init(name: "state", value: UUID().uuidString),
        ]
        let callback = try await webAuth(start: comps.url!)
        let code = URLComponents(url: callback, resolvingAgainstBaseURL: false)!
            .queryItems!.first(where: { $0.name == "code" })!.value!
        return try await exchange(code: code, verifier: verifier)
    }

    private func webAuth(start: URL) async throws -> URL {
        try await withCheckedThrowingContinuation { cont in
            let session = ASWebAuthenticationSession(
                url: start,
                callbackURLScheme: redirect.scheme
            ) { url, error in
                if let url { cont.resume(returning: url) }
                else { cont.resume(throwing: error ?? URLError(.cancelled)) }
            }
            session.presentationContextProvider = self
            session.prefersEphemeralWebBrowserSession = true
            session.start()
        }
    }

    private static func codeVerifier() -> String {
        let data = (0..<32).map { _ in UInt8.random(in: 0...255) }
        return Data(data).base64URLEncodedString()
    }
    private static func codeChallenge(for verifier: String) -> String {
        Data(SHA256.hash(data: Data(verifier.utf8))).base64URLEncodedString()
    }
    func presentationAnchor(for _: ASWebAuthenticationSession) -> ASPresentationAnchor {
        UIApplication.shared.connectedScenes
            .compactMap { ($0 as? UIWindowScene)?.keyWindow }.first!
    }
}
```

`prefersEphemeralWebBrowserSession = true` means no Safari cookies leak in or
out — critical for multi-account UX and regulated environments.

## Refresh tokens without deadlock

The hard part isn't refreshing the token — it's doing it when 50 requests all
401 at once. Every one of them tries to refresh, and now you've burned 50
refresh tokens. The fix is to **serialize and deduplicate** with an actor:

```swift
actor TokenRefresher {
    private var cached: Token?
    private var inFlight: Task<Token, Error>?

    func validToken() async throws -> Token {
        if let token = cached, !token.isExpired { return token }
        if let task = inFlight { return try await task.value }
        let task = Task { try await self.performRefresh() }
        inFlight = task
        defer { inFlight = nil }
        let fresh = try await task.value
        cached = fresh
        return fresh
    }

    private func performRefresh() async throws -> Token {
        let refresh = try Keychain.shared.read(account: "refresh")
        // POST to /token with grant_type=refresh_token&refresh_token=...
        // decode, return
        return try await doHTTPRefresh(refresh: refresh)
    }
}
```

Because `actor` serializes method entry, only one `performRefresh` ever runs.
The other 49 callers `await` the same `inFlight` task and all receive the same
new token. Zero thundering herd.

## Retry with exponential backoff + jitter

```swift
func withRetry<T>(
    maxAttempts: Int = 4,
    baseDelay: TimeInterval = 0.5,
    shouldRetry: (Error) -> Bool = { _ in true },
    _ operation: () async throws -> T
) async throws -> T {
    var lastError: Error?
    for attempt in 0..<maxAttempts {
        do { return try await operation() }
        catch {
            guard shouldRetry(error), attempt < maxAttempts - 1 else { throw error }
            let backoff = baseDelay * pow(2, Double(attempt))
            let jitter = Double.random(in: 0...0.3) * backoff
            try await Task.sleep(for: .seconds(backoff + jitter))
            lastError = error
        }
    }
    throw lastError ?? URLError(.unknown)
}
```

Retry only on **transient** failures: timeouts, 502/503/504, connection lost.
**Never retry** 4xx (except 401 after refresh) — they won't succeed next time.

## Certificate pinning

Pinning prevents a compromised CA (or a corporate MITM proxy) from reading your
traffic. The SwiftCrypto-friendly path — hash the SubjectPublicKeyInfo and
compare:

```swift
final class Pinner: NSObject, URLSessionDelegate {
    let pinnedSPKIHashes: Set<Data>

    func urlSession(
        _ session: URLSession,
        didReceive challenge: URLAuthenticationChallenge,
        completionHandler: @escaping (URLSession.AuthChallengeDisposition, URLCredential?) -> Void
    ) {
        guard
            challenge.protectionSpace.authenticationMethod == NSURLAuthenticationMethodServerTrust,
            let trust = challenge.protectionSpace.serverTrust,
            let cert = (SecTrustCopyCertificateChain(trust) as? [SecCertificate])?.first,
            let key  = SecCertificateCopyKey(cert),
            let spki = SecKeyCopyExternalRepresentation(key, nil) as Data?
        else { return completionHandler(.cancelAuthenticationChallenge, nil) }
        let hash = Data(SHA256.hash(data: spki))
        if pinnedSPKIHashes.contains(hash) {
            completionHandler(.useCredential, URLCredential(trust: trust))
        } else {
            completionHandler(.cancelAuthenticationChallenge, nil)
        }
    }
}
```

Pin at least two keys (current + backup) so you can rotate without an app
update.

<Callout kind="warning" title="Never ship a static Bearer token in your binary">
Anyone with `strings` or Hopper can extract hardcoded credentials from an `.ipa`
in seconds. No API key, OAuth client secret, or admin token should live in your
binary. Use PKCE for user auth, short-lived server-issued tokens for backend
operations, and App Attest for device identity. If an intern adds `let apiKey =`
in a commit — rotate the key and reject the PR.
</Callout>

## Request signing (HMAC)

Some APIs (AWS, Stripe webhooks) require you sign each request with an HMAC
over the method + path + body + timestamp. Stops replay attacks if the signing
key is device-specific.

```swift
import CryptoKit

func sign(_ request: inout URLRequest, secret: SymmetricKey) {
    let ts = String(Int(Date.now.timeIntervalSince1970))
    let method = request.httpMethod ?? "GET"
    let path = request.url?.path ?? ""
    let body = request.httpBody ?? Data()
    var canonical = Data("\(method)\n\(path)\n\(ts)\n".utf8)
    canonical.append(body)
    let mac = HMAC<SHA256>.authenticationCode(for: canonical, using: secret)
    request.setValue(ts, forHTTPHeaderField: "X-Timestamp")
    request.setValue(Data(mac).base64EncodedString(), forHTTPHeaderField: "X-Signature")
}
```

## Apollo iOS for GraphQL

Apollo generates typed Swift from your `.graphql` files so queries aren't strings.

```graphql
# LaunchList.graphql
query LaunchList {
  launches(limit: 20) {
    id
    mission_name
    launch_date_utc
  }
}
```

Run `apollo-ios-cli generate` and you get a typed `LaunchListQuery` struct. Use
it:

```swift
import Apollo

let client = ApolloClient(url: URL(string: "https://api.spacex.land/graphql/")!)

client.fetch(query: LaunchListQuery()) { result in
    switch result {
    case .success(let gql):
        let launches = gql.data?.launches ?? []
        print(launches.compactMap { $0?.missionName })
    case .failure(let error):
        print("GraphQL error: \(error)")
    }
}
```

Full setup: [apollographql.com/docs/ios](https://www.apollographql.com/docs/ios).

## Key takeaways

- **PKCE is the only correct OAuth flow on mobile** — you cannot keep a client secret.
- **Serialize token refresh in an actor** — `inFlight` task dedupes concurrent 401s.
- **Retry only on transient errors** with exponential backoff + jitter.
- **Pin at least two keys** so you can rotate without an app update.
- **Never ship static secrets** — if you did, rotate now.
- **GraphQL pays off when you have many clients** on one schema; otherwise REST is fine.
