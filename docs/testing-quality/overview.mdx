```swift
import Testing

struct CartTests {
    @Test func emptyCartTotalIsZero() {
        let cart = Cart()
        #expect(cart.total == .zero)
    }

    @Test(arguments: [1, 5, 10])
    func addingItemsIncreasesCount(_ count: Int) {
        let cart = Cart()
        for _ in 0..<count { cart.add(Product.sample) }
        #expect(cart.items.count == count)
    }
}
```

## Protocol-driven mocking

The pattern that makes everything testable: depend on protocols, inject in init.

```swift
final class FakeProductRepository: ProductRepository {
    var stub: Result<[Product], Error> = .success([])
    var fetchCallCount = 0
    func fetchAll() async throws -> [Product] {
        fetchCallCount += 1
        return try stub.get()
    }
}

@Test func showsLoadedStateOnSuccess() async {
    let repo  = FakeProductRepository()
    repo.stub = .success([.sample])
    let vm    = ProductListViewModel(useCase: FetchProductsUseCase(repo: repo))
    await vm.load()
    if case .loaded(let p) = vm.state { #expect(p.count == 1) } else { Issue.record() }
}
```

No mocking framework needed — Swift's protocols + value semantics do the job.

## XCUITest in 30 seconds

```swift
final class CheckoutFlowUITests: XCTestCase {
    func testHappyPath() {
        let app = XCUIApplication()
        app.launch()
        app.buttons["addToCart"].tap()
        app.buttons["checkout"].tap()
        XCTAssertTrue(app.staticTexts["Order placed!"].waitForExistence(timeout: 3))
    }
}
```

Tag every interactive element with an `accessibilityIdentifier` — also wins
you accessibility points for free.

## Snapshot testing pins UI regressions

```swift
import SnapshotTesting

@Test func productCardLooksRight() {
    let view = ProductCard(product: .sample)
    assertSnapshot(of: view, as: .image(layout: .device(config: .iPhone13)))
}
```

Run once to record; every subsequent run diffs the image. Visual regressions
fail loudly.

## Key takeaways

- **Swift Testing is the new default** — adopt it for new projects in Xcode 16+.
- **Protocol-driven design = testability** — depend on protocols, inject implementations.
- **XCUITest is for happy paths**, not exhaustive coverage. Lean on unit tests for logic.
- **Snapshot tests catch visual regressions** that pure assertion tests miss.
- **CI from day one** — Xcode Cloud or GitHub Actions, with mandatory PR checks.
